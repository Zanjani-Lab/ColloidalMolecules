# Sample LAMMPS input file to run an NPT relaxation MD simulation for a superstrcuture of Cubic and Octahedral Colloidal clusters
# Author: Zanjani research group: http://cec.miamioh.edu/zanjani-research 
# This input file can be revised to run a variety of simulations, e.g. self-assembly, property calculation, etc., for various types of superstructures of colloidal molecules
# Details about the superstructures are presented in the following reference:
# Aryana, K., Stahley, J. B., Parvez, N., Kim, K. and Zanjani, M. B. (2019) Superstructures of multielement colloidal molecules: efficient pathways to construct reconfigurable photonic and phononic crystals. Adv. Theory Simul. p. 1800198.
# To visualize the configuration, you can load the output dump file into the Ovito (or similar software), and perform an Affine Transformation that scale everything with a factor of 1e5
# Following the previous step in Ovito, you can select the following radii for various types
# raduis of type 1 for visualization in ovito = 2
# raduis of type 2 for visualization in ovito = 3.75
# raduis of type 3 for visualization in ovito = 2
# raduis of type 4 for visualization in ovito = 1.46

# Tables for interaction potentials (pair_coeff commands) are provided under the same GitHub repository.

clear
units           cgs
dimension       3
boundary        p p p
atom_style      bond
atom_modify     sort 10 0.001


read_data CUBEOCT_initLAMMPS_3by3.txt # this read_data file is generated by the provided python code

bond_style      harmonic
bond_coeff	    1 5e-2	4.11e-5	
bond_coeff      2 5e-2  2.91e-5
bond_coeff      3 5e-2  4.11e-5
bond_coeff      4 5e-2  3.56e-5

velocity all create 300 2045

neighbor        2e-5 multi
neigh_modify    delay 0

pair_style     hybrid table linear 601
pair_coeff      1 1 table Rep_400.txt DNA_Rep_Potential
pair_coeff      1 2 table Rep_560.txt DNA_Rep_Potential
pair_coeff      1 3 table Att_400_4p5kbt.txt DNA_Att_Potential
pair_coeff      1 4 none

pair_coeff      2 2 table Rep_720.txt DNA_Rep_Potential
pair_coeff      2 3 table Rep_560.txt DNA_Rep_Potential
pair_coeff      2 4 none
pair_coeff      3 3 table Rep_400.txt DNA_Rep_Potential
pair_coeff      3 4 none
pair_coeff      4 4 none



#Initial Relaxation- sample run

fix 1 all npt temp 300.0 300.0 0.5e-7 iso 0.0 0.0 0.5e-6 

timestep        0.5e-9


thermo_style custom step temp pe etotal fmax lx ly lz

dump           8 all custom 10000 outCUBEOCT_initial id type x y z
dump_modify     8 sort id

thermo 100000

comm_modify cutoff 12e-5
run 500000
